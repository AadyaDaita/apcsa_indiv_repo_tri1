public int getChargingCost(int chargeTime){
    int minimum_cost = 0;
    int start_time = 0;
    
    for (int hour=0; hour<24; hour++)
    {
        int total_rate = 0;
        int curr_hour = hour;
        for (int i=0; i < chargeTime; i++){

            if( (curr_hour + i) > 23)
            {
                curr_hour = 0;
                chargeTime -= i;
                i = 0;
            }
            total_rate += rateTable[curr_hour + i];
        }

        if( (minimum_cost == 0) || (total_rate < minimum_cost) )
        {
            minimum_cost = total_rate;
            start_time = hour;
        }
}

return start_time;

}
